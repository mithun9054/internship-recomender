  </select>

  <label for="sector">Sector Interest</label>
  <select id="sector">
    <option value="IT">IT</option>
    <option value="Education">Education</option>
    <option value="Health">Health</option>
  </select>

  <label for="location">Location Preference</label>
  <select id="location">
    <option value="Delhi">Delhi</option>
    <option value="Tamil Nadu">Tamil Nadu</option>
    <option value="Remote">Remote</option>
  </select>

  <button onclick="recommend()">Get Recommendations</button>

  <div id="results"></div>

  <script>
    const internships = [
      { title: "Python Developer Intern", sector: "IT", skill: "Python", location: "Remote" },
      { title: "Education Outreach Intern", sector: "Education", skill: "Marketing", location: "Tamil Nadu" },
      { title: "Health Data Entry Intern", sector: "Health", skill: "Data Entry", location: "Delhi" },
      { title: "Digital Literacy Intern", sector: "Education", skill: "Data Entry", location: "Remote" },
      { title: "AI Research Intern", sector: "IT", skill: "Python", location: "Delhi" }
    ];

    function recommend() {
      const skill = document.getElementById("skills").value;
      const sector = document.getElementById("sector").value;
      const location = document.getElementById("location").value;

      const matches = internships.filter(i =>
        i.skill === skill || i.sector === sector || i.location === location
      ).slice(0, 5);

      const resultsDiv = document.getElementById("results");
      resultsDiv.innerHTML = "<h3>Recommended Internships:</h3>";
      matches.forEach(i => {
        resultsDiv.innerHTML += `<div class="card">
          <strong>${i.title}</strong><br>
          Sector: ${i.sector}<br>
          Location: ${i.location}<br>
          Skill Match: ${i.skill}
        </div>`;
      });
    }
  </script>
